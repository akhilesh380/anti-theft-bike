#include <Keypad.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>

// LCD setup
LiquidCrystal_I2C lcd(0x27, 16, 2);

// Keypad setup
const byte ROWS = 4;
const byte COLS = 4;
char keys[ROWS][COLS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};

byte rowPins[ROWS] = {2, 3, 4, 5};   // R1, R2, R3, R4
byte colPins[COLS] = {6, 7, 8, 9};   // C1, C2, C3, C4

Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

// Output pins
const int relayPin = 12;
const int buzzerPin = 10;

// Password
String inputCode = "";
const String correctCode = "3104";
int wrongAttempts = 0;
bool relayState = false;

void setup() {
  pinMode(relayPin, OUTPUT);
  pinMode(buzzerPin, OUTPUT);
  digitalWrite(relayPin, LOW);
  digitalWrite(buzzerPin, LOW);

  lcd.begin();
  lcd.backlight();
  lcd.setCursor(0, 0);
  lcd.print("Enter Password:");
  lcd.setCursor(0, 1);

  Serial.begin(9600);
}

void loop() {
  char key = keypad.getKey();

  if (key) {
    Serial.print("Key pressed: ");
    Serial.println(key);

    if (key == '#') {
      // Check password
      if (inputCode == correctCode) {
        relayState = true;
        digitalWrite(relayPin, HIGH);
        lcd.clear();
        lcd.setCursor(0, 0);
        lcd.print("Access Granted");
        Serial.println("‚úÖ Correct Password");
        wrongAttempts = 0;  // reset on success
      } else {
        wrongAttempts++;
        lcd.clear();
        lcd.setCursor(0, 0);
        lcd.print("Wrong Password!");
        Serial.print("‚ùå Attempt: ");
        Serial.println(wrongAttempts);

        // Activate buzzer after 3 wrong tries
        if (wrongAttempts >= 3) {
          lcd.setCursor(0, 1);
          lcd.print("ALARM TRIGGERED!");
          digitalWrite(buzzerPin, HIGH); // Turn on buzzer
          Serial.println("üö® Buzzer ON: 3 wrong attempts");
        }
      }

      delay(2000);
      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("Enter Password:");
      inputCode = "";
      lcd.setCursor(0, 1);
    } 
    else if (key == '*') {
      // Reset everything
      inputCode = "";
      relayState = false;
      wrongAttempts = 0;
      digitalWrite(relayPin, LOW);
      digitalWrite(buzzerPin, LOW);
      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("System Reset");
      Serial.println("üîÅ System Reset");
      delay(2000);
      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("Enter Password:");
      lcd.setCursor(0, 1);
    } 
    else {
      // Add key to password input
      if (inputCode.length() < 16) {
        inputCode += key;
        lcd.print("*");
      }
    }
  }
}
